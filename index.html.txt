  <script>
    function showMessages() {
      document.getElementById('messagesSection').style.display = 'block';
    }

    function editMessages() {
      let newMessage = prompt("Digite uma nova mensagem para a Etapa 1:");
      if (newMessage) {
        const ul = document.querySelector('.message-list');
        const li = document.createElement('li');
        li.innerHTML = "<strong>Etapa 1:</strong> " + newMessage;
        ul.appendChild(li);
      }
    }
  </script>
